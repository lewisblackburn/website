## Installation

### With NPM

> `npm i --save use-count`

### Or, with yarn:

> `yarn add use-count`

## Usage

```tsx
import { useCount } from 'use-count'

const App = () => {
  const { value: hits } = useCount('namespace', 'key')

  return <p>{hits}</p>
}
```
